### 1. Notification Service Architecture
```javascript
// Notification Service Structure
class NotificationService {
  constructor() {
    this.providers = {
      email: new EmailProvider(),
      push: new PushNotificationProvider(),
      sms: new SMSProvider()
    };
    this.templateEngine = new TemplateEngine();
    this.deliveryTracker = new DeliveryTracker();
    this.subscriptionManager = new SubscriptionManager();
  }
  
  async sendNotification(notificationData) {
    try {
      // Validate notification data
      await this.validateNotification(notificationData);
      
      // Get user preferences
      const preferences = await this.getUserPreferences(notificationData.userId);
      
      // Check if user wants this type of notification
      if (!this.shouldSendNotification(notificationData.type, preferences)) {
        return { sent: false, reason: 'User opted out' };
      }
      
      // Process notification based on type
      const result = await this.processNotification(notificationData);
      
      // Track delivery
      await this.deliveryTracker.track(notificationData, result);
      
      return result;
    } catch (error) {
      await this.logError(notificationData, error);
      throw error;
    }
  }
  
  async processNotification(notificationData) {
    const { type, userId, data } = notificationData;
    
    switch (type) {
      case 'email':
        return await this.providers.email.send(notificationData);
      case 'push':
        return await this.providers.push.send(notificationData);
      case 'sms':
        return await this.providers.sms.send(notificationData);
      case 'multi':
        return await this.sendMultiChannel(notificationData);
      default:
        throw new Error('Unsupported notification type');
    }
  }
}
```

### 2. Email Provider
```javascript
const nodemailer = require('nodemailer');
const { Resend } = require('resend');

class EmailProvider {
  constructor() {
    this.transporter = nodemailer.createTransporter({
      host: process.env.SMTP_HOST,
      port: process.env.SMTP_PORT,
      secure: process.env.SMTP_SECURE === 'true',
      auth: {
        user: process.env.SMTP_USER,
        pass: process.env.SMTP_PASSWORD
      }
    });
    
    this.resend = new Resend(process.env.RESEND_API_KEY);
  }
  
  async send(notificationData) {
    try {
      const { userId, template, data, to, subject } = notificationData;
      
      // Get user email if not provided
      const email = to || await this.getUserEmail(userId);
      
      // Render template
      const htmlContent = await this.templateEngine.render(template, data);
      const textContent = await this.templateEngine.renderText(template, data);
      
      // Send email
      const result = await this.transporter.sendMail({
        from: process.env.FROM_EMAIL,
        to: email,
        subject: subject || await this.getTemplateSubject(template),
        html: htmlContent,
        text: textContent,
        headers: {
          'X-Notification-Type': notificationData.type,
          'X-User-ID': userId
        }
      });
      
      return {
        success: true,
        messageId: result.messageId,
        provider: 'smtp',
        deliveredAt: new Date()
      };
    } catch (error) {
      // Fallback to Resend
      return await this.sendViaResend(notificationData);
    }
  }
  
  async sendViaResend(notificationData) {
    try {
      const { userId, template, data, to, subject } = notificationData;
      
      const email = to || await this.getUserEmail(userId);
      const htmlContent = await this.templateEngine.render(template, data);
      
      const result = await this.resend.emails.send({
        from: process.env.FROM_EMAIL,
        to: email,
        subject: subject || await this.getTemplateSubject(template),
        html: htmlContent
      });
      
      return {
        success: true,
        messageId: result.id,
        provider: 'resend',
        deliveredAt: new Date()
      };
    } catch (error) {
      throw new NotificationError('Email sending failed', error);
    }
  }
  
  async getUserEmail(userId) {
    const user = await User.findById(userId);
    if (!user) {
      throw new Error('User not found');
    }
    return user.email;
  }
}
```

### 3. Push Notification Provider
```javascript
const admin = require('firebase-admin');

class PushNotificationProvider {
  constructor() {
    this.firebase = admin.initializeApp({
      credential: admin.credential.cert({
        projectId: process.env.FIREBASE_PROJECT_ID,
        clientEmail: process.env.FIREBASE_CLIENT_EMAIL,
        privateKey: process.env.FIREBASE_PRIVATE_KEY.replace(/\\n/g, '\n')
      })
    });
  }
  
  async send(notificationData) {
    try {
      const { userId, title, body, data, imageUrl } = notificationData;
      
      // Get user's FCM tokens
      const tokens = await this.getUserTokens(userId);
      
      if (tokens.length === 0) {
        return { success: false, reason: 'No FCM tokens found' };
      }
      
      // Prepare message
      const message = {
        notification: {
          title: title || 'Notification',
          body: body || 'You have a new notification',
          imageUrl: imageUrl
        },
        data: data || {},
        tokens: tokens
      };
      
      // Send notification
      const result = await this.firebase.messaging().sendMulticast(message);
      
      // Handle results
      const successfulTokens = [];
      const failedTokens = [];
      
      result.responses.forEach((response, index) => {
        if (response.success) {
          successfulTokens.push(tokens[index]);
        } else {
          failedTokens.push({
            token: tokens[index],
            error: response.error
          });
        }
      });
      
      // Remove invalid tokens
      if (failedTokens.length > 0) {
        await this.removeInvalidTokens(failedTokens);
      }
      
      return {
        success: result.successCount > 0,
        successCount: result.successCount,
        failureCount: result.failureCount,
        successfulTokens,
        failedTokens
      };
    } catch (error) {
      throw new NotificationError('Push notification failed', error);
    }
  }
  
  async getUserTokens(userId) {
    const user = await User.findById(userId);
    if (!user || !user.fcmTokens) {
      return [];
    }
    
    // Filter out expired tokens
    const validTokens = user.fcmTokens.filter(token => 
      token.expiresAt > new Date()
    );
    
    return validTokens.map(token => token.value);
  }
  
  async removeInvalidTokens(failedTokens) {
    for (const failedToken of failedTokens) {
      await User.updateMany(
        { 'fcmTokens.value': failedToken.token },
        { $pull: { fcmTokens: { value: failedToken.token } } }
      );
    }
  }
}
```

```javascript
const twilio = require('twilio');

class SMSProvider {
  constructor() {
    this.client = twilio(
      process.env.TWILIO_ACCOUNT_SID,
      process.env.TWILIO_AUTH_TOKEN
    );
  }
  
  async send(notificationData) {
    try {

      });
      
      return {
        success: true,
        messageId: result.sid,
        provider: 'twilio',
        deliveredAt: new Date()

      throw new NotificationError('SMS sending failed', error);
```

### 5. Template Engine
```javascript
const handlebars = require('handlebars');
const fs = require('fs').promises;
const path = require('path');

class TemplateEngine {
  constructor() {
    this.templates = new Map();
    this.loadTemplates();
  }
  
  async loadTemplates() {
    const templatesDir = path.join(__dirname, 'templates');
    const files = await fs.readdir(templatesDir);
    
    for (const file of files) {
      if (file.endsWith('.hbs')) {
        const templateName = path.basename(file, '.hbs');
        const templateContent = await fs.readFile(
          path.join(templatesDir, file),
          'utf8'
        );
        
        this.templates.set(templateName, handlebars.compile(templateContent));
      }
    }
  }
  
  async render(templateName, data) {
    const template = this.templates.get(templateName);
    if (!template) {
      throw new Error(`Template ${templateName} not found`);
    }
    
    return template(data);
  }
  
  async renderText(templateName, data) {
    // Convert HTML to text
    const html = await this.render(templateName, data);
    return this.htmlToText(html);
  }
  
  htmlToText(html) {
    // Simple HTML to text conversion
    return html
      .replace(/<[^>]*>/g, '')
      .replace(/\s+/g, ' ')
      .trim();
  }
}
```

### 7. Subscription Management
```javascript
class SubscriptionManager {
  async getUserPreferences(userId) {
    const user = await User.findById(userId);
    if (!user) {
      throw new Error('User not found');
    }
    
    return {
      email: user.notificationPreferences?.email || true,
      push: user.notificationPreferences?.push || true,
      sms: user.notificationPreferences?.sms || false,
      types: user.notificationPreferences?.types || {
        payment: true,
        security: true,
        marketing: false
      }
    };
  }
  
  async updatePreferences(userId, preferences) {
    await User.findByIdAndUpdate(userId, {
      $set: {
        notificationPreferences: preferences
      }
    });
  }
  
  async shouldSendNotification(type, preferences) {
    // Check if user wants this type of notification
    if (!preferences[type]) {
      return false;
    }
    
    // Check notification type preferences
    const notificationType = this.getNotificationType(type);
    if (notificationType && !preferences.types[notificationType]) {
      return false;
    }
    
    return true;
  }
  
  getNotificationType(notificationType) {
    const typeMap = {
      'payment_confirmation': 'payment',
      'payment_failed': 'payment',
      'security_alert': 'security',
      'marketing_promo': 'marketing'
    };
    
    return typeMap[notificationType] || 'general';
  }
}
```

### 8. Delivery Tracking
```javascript
class DeliveryTracker {
  async track(notificationData, result) {
    const trackingRecord = new NotificationTracking({
      userId: notificationData.userId,
      type: notificationData.type,
      template: notificationData.template,
      status: result.success ? 'sent' : 'failed',
      provider: result.provider,
      messageId: result.messageId,
      sentAt: new Date(),
      data: notificationData.data
    });
    
    await trackingRecord.save();
    
    // Set up delivery status checking for email
    if (notificationData.type === 'email' && result.success) {
      this.scheduleDeliveryCheck(result.messageId, trackingRecord._id);
    }
  }
  
  async scheduleDeliveryCheck(messageId, trackingId) {
    // Schedule delivery status check
    setTimeout(async () => {
      await this.checkDeliveryStatus(messageId, trackingId);
    }, 30000); // Check after 30 seconds
  }
  
  async checkDeliveryStatus(messageId, trackingId) {
    try {
      // Check delivery status with provider
      const status = await this.getProviderDeliveryStatus(messageId);
      
      await NotificationTracking.findByIdAndUpdate(trackingId, {
        $set: {
          deliveryStatus: status.status,
          deliveredAt: status.deliveredAt,
          lastChecked: new Date()
        }
      });
    } catch (error) {
      console.error('Delivery status check failed:', error);
    }
  }
}
```

### 9. Notification Queue
```javascript
const Bull = require('bull');
const Redis = require('ioredis');

class NotificationQueue {
  constructor() {
    this.redis = new Redis(process.env.REDIS_URL);
    this.emailQueue = new Bull('email notifications', { redis: this.redis });
    this.pushQueue = new Bull('push notifications', { redis: this.redis });
    this.smsQueue = new Bull('sms notifications', { redis: this.redis });
    
    this.setupProcessors();
  }
  
  setupProcessors() {
    // Email processor
    this.emailQueue.process('send-email', async (job) => {
      const { notificationData } = job.data;
      const emailProvider = new EmailProvider();
      return await emailProvider.send(notificationData);
    });
    
    // Push notification processor
    this.pushQueue.process('send-push', async (job) => {
      const { notificationData } = job.data;
      const pushProvider = new PushNotificationProvider();
      return await pushProvider.send(notificationData);
    });
    
    // SMS processor
    this.smsQueue.process('send-sms', async (job) => {
      const { notificationData } = job.data;
      const smsProvider = new SMSProvider();
      return await smsProvider.send(notificationData);
    });
  }
  
  async addEmailNotification(notificationData, options = {}) {
    return await this.emailQueue.add('send-email', {
      notificationData
    }, {
      delay: options.delay || 0,
      attempts: options.attempts || 3,
      backoff: options.backoff || 'exponential'
    });
  }
  
  async addPushNotification(notificationData, options = {}) {
    return await this.pushQueue.add('send-push', {
      notificationData
    }, {
      delay: options.delay || 0,
      attempts: options.attempts || 3
    });
  }
  
  async addSMSNotification(notificationData, options = {}) {
    return await this.smsQueue.add('send-sms', {
      notificationData
    }, {
      delay: options.delay || 0,
      attempts: options.attempts || 3
    });
  }
}
```
