### 1. Payment Service Architecture
```javascript
// Payment Service Structure
class PaymentService {
  constructor() {
    this.providers = {
      stripe: new StripeProvider(),
      paypal: new PayPalProvider(),
      crypto: new CryptoProvider(),
      bank: new BankTransferProvider()
    };
    this.validator = new PaymentValidator();
    this.encryptor = new PaymentEncryptor();
    this.auditor = new PaymentAuditor();
  }
  
  async processPayment(paymentData) {
    try {
      // Validate payment data
      await this.validator.validate(paymentData);
      
      // Encrypt sensitive data
      const encryptedData = await this.encryptor.encrypt(paymentData);
      
      // Process based on payment method
      const result = await this.routePayment(encryptedData);
      
      // Audit transaction
      await this.auditor.logTransaction(paymentData, result);
      
      return result;
    } catch (error) {
      await this.auditor.logError(paymentData, error);
      throw error;
    }
  }
  
  async routePayment(paymentData) {
    const method = paymentData.paymentMethod;
    
    switch (method) {
      case 'card':
        return await this.providers.stripe.process(paymentData);
      case 'paypal':
        return await this.providers.paypal.process(paymentData);
      case 'crypto':
        return await this.providers.crypto.process(paymentData);
      case 'bank':
        return await this.providers.bank.process(paymentData);
      default:
        throw new Error('Unsupported payment method');
    }
  }
}
```

### 3. Payment Validation
```javascript
class PaymentValidator {
  validate(paymentData) {
    const errors = [];
    
    // Amount validation
    if (!paymentData.amount || paymentData.amount <= 0) {
      errors.push('Amount must be greater than 0');
    }
    
    if (paymentData.amount > 100000) {
      errors.push('Amount exceeds maximum limit');
    }
    
    // Currency validation
    const supportedCurrencies = ['USD', 'EUR', 'GBP', 'BTC', 'ETH'];
    if (!supportedCurrencies.includes(paymentData.currency)) {
      errors.push('Unsupported currency');
    }
    
    // Payment method validation
    const supportedMethods = ['card', 'paypal', 'crypto', 'bank_transfer'];
    if (!supportedMethods.includes(paymentData.paymentMethod)) {
      errors.push('Unsupported payment method');
    }
    
    // Method-specific validation
    switch (paymentData.paymentMethod) {
      case 'card':
        this.validateCardData(paymentData, errors);
        break;
      case 'crypto':
        this.validateCryptoData(paymentData, errors);
        break;
      case 'bank_transfer':
        this.validateBankData(paymentData, errors);
        break;
    }
    
    if (errors.length > 0) {
      throw new ValidationError('Payment validation failed', errors);
    }
  }
  
  validateCardData(paymentData, errors) {
    const { cardNumber, expiryMonth, expiryYear, cvv } = paymentData;
    
    if (!cardNumber || !this.isValidCardNumber(cardNumber)) {
      errors.push('Invalid card number');
    }
    
    if (!expiryMonth || expiryMonth < 1 || expiryMonth > 12) {
      errors.push('Invalid expiry month');
    }
    
    if (!expiryYear || expiryYear < new Date().getFullYear()) {
      errors.push('Invalid expiry year');
    }
    
    if (!cvv || !this.isValidCVV(cvv)) {
      errors.push('Invalid CVV');
    }
  }
  
  validateCryptoData(paymentData, errors) {
    const { cryptoAddress, network } = paymentData;
    
    if (!cryptoAddress || !this.isValidCryptoAddress(cryptoAddress, network)) {
      errors.push('Invalid crypto address');
    }
    
    if (!network || !['bitcoin', 'ethereum', 'litecoin'].includes(network)) {
      errors.push('Unsupported crypto network');
    }
  }
  
  isValidCardNumber(cardNumber) {
    // Luhn algorithm validation
    const digits = cardNumber.replace(/\D/g, '');
    let sum = 0;
    let isEven = false;
    
    for (let i = digits.length - 1; i >= 0; i--) {
      let digit = parseInt(digits[i]);
      
      if (isEven) {
        digit *= 2;
        if (digit > 9) {
          digit -= 9;
        }
      }
      
      sum += digit;
      isEven = !isEven;
    }
    
    return sum % 10 === 0;
  }
}
```

```javascript
const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);

class StripeProvider {
  async process(paymentData) {
    try {
      // Create payment intent
      const paymentIntent = await stripe.paymentIntents.create({
        amount: Math.round(paymentData.amount * 100), // Convert to cents
        currency: paymentData.currency.toLowerCase(),
        payment_method: paymentData.paymentMethodId,
        confirmation_method: 'manual',
        confirm: true,
        metadata: {
          userId: paymentData.userId,
          description: paymentData.description
        }
      });
      
      return {
        success: paymentIntent.status === 'succeeded',
        transactionId: paymentIntent.id,
        status: paymentIntent.status,
        provider: 'stripe',
        providerResponse: paymentIntent
      };
    } catch (error) {
      throw new PaymentError('Stripe payment failed', error);
    }
  }
  
  async createPaymentMethod(cardData) {
    try {
      const paymentMethod = await stripe.paymentMethods.create({
        type: 'card',
        card: {
          number: cardData.cardNumber,
          exp_month: cardData.expiryMonth,
          exp_year: cardData.expiryYear,
          cvc: cardData.cvv
        }
      });
      
      return paymentMethod.id;
    } catch (error) {
      throw new PaymentError('Failed to create payment method', error);
    }
  }
  
  async refund(transactionId, amount) {
    try {
      const refund = await stripe.refunds.create({
        payment_intent: transactionId,
        amount: amount ? Math.round(amount * 100) : undefined
      });
      
      return {
        success: true,
        refundId: refund.id,
        status: refund.status
      };
    } catch (error) {
      throw new PaymentError('Refund failed', error);
    }
  }
}
```

```javascript
const { ethers } = require('ethers');

class CryptoProvider {
  constructor() {
    this.provider = new ethers.providers.JsonRpcProvider(process.env.ETHEREUM_RPC_URL);
    this.wallet = new ethers.Wallet(process.env.PRIVATE_KEY, this.provider);
  }
  
  async process(paymentData) {
    try {
      const { amount, currency, toAddress } = paymentData;
      
      // Convert amount to wei
      const amountWei = ethers.utils.parseEther(amount.toString());
      
      // Create transaction
      const tx = await this.wallet.sendTransaction({
        to: toAddress,
        value: amountWei,
        gasLimit: 21000
      });
      
      // Wait for confirmation
      const receipt = await tx.wait();
      
      return {
        success: receipt.status === 1,
        transactionId: tx.hash,
        status: receipt.status === 1 ? 'completed' : 'failed',
        provider: 'crypto',
        confirmations: receipt.confirmations,
        blockNumber: receipt.blockNumber
      };
    } catch (error) {
      throw new PaymentError('Crypto payment failed', error);
    }
  }
  
  async getBalance(address) {
    try {
      const balance = await this.provider.getBalance(address);
      return ethers.utils.formatEther(balance);
    } catch (error) {
      throw new PaymentError('Failed to get balance', error);
    }
  }
  
  async getTransactionStatus(txHash) {
    try {
      const tx = await this.provider.getTransaction(txHash);
      const receipt = await this.provider.getTransactionReceipt(txHash);
      
      return {
        hash: txHash,
        status: receipt ? (receipt.status === 1 ? 'completed' : 'failed') : 'pending',
        confirmations: receipt ? receipt.confirmations : 0,
        blockNumber: receipt ? receipt.blockNumber : null
      };
    } catch (error) {
      throw new PaymentError('Failed to get transaction status', error);
    }
  }
}
```

### 5. Payment Processing Flow
```javascript
class PaymentProcessor {
  constructor() {
    this.paymentService = new PaymentService();
    this.notificationService = new NotificationService();
    this.auditService = new AuditService();
  }
  
  async processPayment(paymentData) {
    const startTime = Date.now();
    
    try {
      // Create payment record
      const payment = new Payment({
        userId: paymentData.userId,
        amount: paymentData.amount,
        currency: paymentData.currency,
        paymentMethod: paymentData.paymentMethod,
        description: paymentData.description,
        status: 'pending',
        metadata: {
          ipAddress: paymentData.ipAddress,
          userAgent: paymentData.userAgent,
          deviceId: paymentData.deviceId
        }
      });
      
      await payment.save();
      
      // Process payment
      const result = await this.paymentService.processPayment(paymentData);
      
      // Update payment status
      payment.status = result.success ? 'completed' : 'failed';
      payment.providerData = {
        transactionId: result.transactionId,
        provider: result.provider,
        providerResponse: result.providerResponse
      };
      payment.processedAt = new Date();
      
      if (result.success) {
        payment.completedAt = new Date();
      } else {
        payment.failedAt = new Date();
      }
      
      await payment.save();
      
      // Send notifications
      await this.notificationService.sendPaymentNotification(payment);
      
      // Audit transaction
      await this.auditService.auditPayment(payment, result);
      
      const processingTime = Date.now() - startTime;
      
      return {
        paymentId: payment._id,
        status: payment.status,
        transactionId: result.transactionId,
        processingTime
      };
      
    } catch (error) {
      // Update payment status to failed
      if (payment) {
        payment.status = 'failed';
        payment.failedAt = new Date();
        await payment.save();
      }
      
      // Log error
      await this.auditService.logError(paymentData, error);
      
      throw error;
    }
  }
  
  async getPaymentStatus(paymentId) {
    const payment = await Payment.findById(paymentId);
    if (!payment) {
      throw new Error('Payment not found');
    }
    
    return {
      paymentId: payment._id,
      status: payment.status,
      amount: payment.amount,
      currency: payment.currency,
      createdAt: payment.createdAt,
      processedAt: payment.processedAt,
      completedAt: payment.completedAt
    };
  }
  
  async getPaymentHistory(userId, filters = {}) {
    const query = { userId };
    
    if (filters.status) {
      query.status = filters.status;
    }
    
    if (filters.dateFrom || filters.dateTo) {
      query.createdAt = {};
      if (filters.dateFrom) {
        query.createdAt.$gte = new Date(filters.dateFrom);
      }
      if (filters.dateTo) {
        query.createdAt.$lte = new Date(filters.dateTo);
      }
    }
    
    const payments = await Payment.find(query)
      .sort({ createdAt: -1 })
      .limit(filters.limit || 50)
      .skip(filters.offset || 0);
    
    return payments;
  }
}
```

### 6. Refund Processing
```javascript
class RefundProcessor {
  async processRefund(paymentId, amount, reason) {
    try {
      const payment = await Payment.findById(paymentId);
      if (!payment) {
        throw new Error('Payment not found');
      }
      
      if (payment.status !== 'completed') {
        throw new Error('Only completed payments can be refunded');
      }
      
      // Create refund record
      const refund = new Refund({
        paymentId: payment._id,
        userId: payment.userId,
        amount: amount || payment.amount,
        reason: reason,
        status: 'pending'
      });
      
      await refund.save();
      
      // Process refund based on payment method
      let result;
      switch (payment.paymentMethod) {
        case 'card':
          result = await this.processStripeRefund(payment, refund);
          break;
        case 'crypto':
          result = await this.processCryptoRefund(payment, refund);
          break;
        default:
          throw new Error('Refund not supported for this payment method');
      }
      
      // Update refund status
      refund.status = result.success ? 'completed' : 'failed';
      refund.transactionId = result.transactionId;
      refund.processedAt = new Date();
      
      if (result.success) {
        refund.completedAt = new Date();
      } else {
        refund.failedAt = new Date();
      }
      
      await refund.save();
      
      // Update payment status
      payment.status = 'refunded';
      payment.refundedAt = new Date();
      await payment.save();
      
      return {
        refundId: refund._id,
        status: refund.status,
        amount: refund.amount,
        transactionId: refund.transactionId
      };
      
    } catch (error) {
      throw new PaymentError('Refund processing failed', error);
    }
  }
  
  async processStripeRefund(payment, refund) {
    try {
      const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);
      
      const stripeRefund = await stripe.refunds.create({
        payment_intent: payment.providerData.transactionId,
        amount: Math.round(refund.amount * 100)
      });
      
      return {
        success: true,
        transactionId: stripeRefund.id,
        status: stripeRefund.status
      };
    } catch (error) {
      return {
        success: false,
        error: error.message
      };
    }
  }
}
```

### 7. Security Implementation
```javascript
class PaymentSecurity {
  constructor() {
    this.encryptor = new PaymentEncryptor();
    this.validator = new PaymentValidator();
    this.fraudDetector = new FraudDetector();
  }
  
  async encryptCardData(cardData) {
    const crypto = require('crypto');
    const algorithm = 'aes-256-gcm';
    const key = crypto.scryptSync(process.env.ENCRYPTION_KEY, 'salt', 32);
    const iv = crypto.randomBytes(16);
    
    const cipher = crypto.createCipher(algorithm, key);
    cipher.setAAD(Buffer.from('card-data', 'utf8'));
    
    let encrypted = cipher.update(JSON.stringify(cardData), 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    return {
      encrypted,
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex')
    };
  }
  
  async detectFraud(paymentData) {
    const riskFactors = [];
    
    // Check for unusual amounts
    if (paymentData.amount > 10000) {
      riskFactors.push('high_amount');
    }
    
    // Check for rapid successive payments
    const recentPayments = await Payment.find({
      userId: paymentData.userId,
      createdAt: { $gte: new Date(Date.now() - 60000) } // Last minute
    });
    
    if (recentPayments.length > 5) {
      riskFactors.push('rapid_payments');
    }
    
    // Check for unusual locations
    if (paymentData.metadata.location) {
      const userLocation = await this.getUserLocation(paymentData.userId);
      if (userLocation && this.isUnusualLocation(userLocation, paymentData.metadata.location)) {
        riskFactors.push('unusual_location');
      }
    }
    
    // Check for known fraud patterns
    if (await this.isKnownFraudPattern(paymentData)) {
      riskFactors.push('known_fraud_pattern');
    }
    
    return {
      isFraud: riskFactors.length > 2,
      riskFactors,
      riskScore: riskFactors.length * 25
    };
  }
  
  async validatePaymentMethod(paymentData) {
    // Validate card data
    if (paymentData.paymentMethod === 'card') {
      return await this.validateCard(paymentData);
    }
    
    // Validate crypto address
    if (paymentData.paymentMethod === 'crypto') {
      return await this.validateCryptoAddress(paymentData);
    }
    
    return true;
  }
}
```
